36 void RiskAnalysis::Analyze() noexcept {
37  assert(results_.empty() && "Rerunning the analysis.");
38  // Set the seed for the pseudo-random number generator if given explicitly.
39  // Otherwise it defaults to the implementation dependent value.
40  if (Analysis::settings().seed() >= 0)
41  Random::seed(Analysis::settings().seed());
42 
43  for (const mef::InitiatingEventPtr& initiating_event :
44  model_->initiating_events()) {
45  if (initiating_event->event_tree()) {
46  LOG(INFO) << "Running event tree analysis: " << initiating_event->name();
47  auto eta = std::make_unique<EventTreeAnalysis>(*initiating_event,
48  Analysis::settings(),
49  model_->context());
50  eta->Analyze();
51  for (EventTreeAnalysis::Result& result : eta->sequences()) {
52  const mef::Sequence& sequence = result.sequence;
53  LOG(INFO) << "Running analysis for sequence: " << sequence.name();
54  results_.push_back(
55  {std::pair<const mef::InitiatingEvent&, const mef::Sequence&>{
56  *initiating_event, sequence}});
57  RunAnalysis(*result.gate, &results_.back());
58  if (result.is_expression_only) {
59  results_.back().fault_tree_analysis = nullptr;
60  results_.back().importance_analysis = nullptr;
61  }
62  if (Analysis::settings().probability_analysis())
63  result.p_sequence = results_.back().probability_analysis->p_total();
64  LOG(INFO) << "Finished analysis for sequence: " << sequence.name();
65  }
66  event_tree_results_.push_back(std::move(eta));
67  LOG(INFO) << "Finished event tree analysis: " << initiating_event->name();
68  }
69  }
70 
71  for (const mef::FaultTreePtr& ft : model_->fault_trees()) {
72  for (const mef::Gate* target : ft->top_events()) {
73  LOG(INFO) << "Running analysis for gate: " << target->id();
74  results_.push_back({target});
75  RunAnalysis(*target, &results_.back());
76  LOG(INFO) << "Finished analysis for gate: " << target->id();
77  }
78  }
79 }

81 void RiskAnalysis::RunAnalysis(const mef::Gate& target,
82  Result* result) noexcept {
83  switch (Analysis::settings().algorithm()) {
84  case Algorithm::kBdd:
85  return RunAnalysis<Bdd>(target, result);
86  case Algorithm::kZbdd:
87  return RunAnalysis<Zbdd>(target, result);
88  case Algorithm::kMocus:
89  return RunAnalysis<Mocus>(target, result);
90  }
91 }

93 template <class Algorithm>
94 void RiskAnalysis::RunAnalysis(const mef::Gate& target,
95  Result* result) noexcept {
96  auto fta = std::make_unique<FaultTreeAnalyzer<Algorithm>>(
97  target, Analysis::settings());
98  fta->Analyze();
99  if (Analysis::settings().probability_analysis()) {
100  switch (Analysis::settings().approximation()) {
101  case Approximation::kNone:
102  RunAnalysis<Algorithm, Bdd>(fta.get(), result);
103  break;
104  case Approximation::kRareEvent:
105  RunAnalysis<Algorithm, RareEventCalculator>(fta.get(), result);
106  break;
107  case Approximation::kMcub:
108  RunAnalysis<Algorithm, McubCalculator>(fta.get(), result);
109  }
110  }
111  result->fault_tree_analysis = std::move(fta);
112 }

114 template <class Algorithm, class Calculator>
115 void RiskAnalysis::RunAnalysis(FaultTreeAnalyzer<Algorithm>* fta,
116  Result* result) noexcept {
117  auto pa = std::make_unique<ProbabilityAnalyzer<Calculator>>(
118  fta, &model_->mission_time());
119  pa->Analyze();
120  if (Analysis::settings().importance_analysis()) {
121  auto ia = std::make_unique<ImportanceAnalyzer<Calculator>>(pa.get());
122  ia->Analyze();
123  result->importance_analysis = std::move(ia);
124  }
125  if (Analysis::settings().uncertainty_analysis()) {
126  auto ua = std::make_unique<UncertaintyAnalyzer<Calculator>>(pa.get());
127  ua->Analyze();
128  result->uncertainty_analysis = std::move(ua);
129  }
130  result->probability_analysis = std::move(pa);
131 }
